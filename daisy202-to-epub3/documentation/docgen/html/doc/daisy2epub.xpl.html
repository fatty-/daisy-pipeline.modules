daisy2epub.xpl

p:declare-step

Transforms a DAISY 2.02-book into an EPUB 3-book.

About

Author: Jostein Austvik Jacobsen (josteinaj@gmail.com), NLB

Maintainer: Jostein Austvik Jacobsen

Programming languageXProc
Version1.0
XProc version1.0

Options

namerequireddescription
href true
output true

These dependencies are derived from the p:import statements from the XProc script.

d2e:ncc - ncc

Load the DAISY 2.02 NCC.

Options

namerequiredconnectiondescription
href inline

Source Code

<d2e:ncc name="ncc" xmlns:p="http://www.w3.org/ns/xproc" xmlns:d2e="http://pipeline.daisy.org/ns/daisy2epub/">
        <p:with-option name="href" select="p:resolve-uri($href)">
            <p:inline>
                <irrelevant/>
            </p:inline>
        </p:with-option>
    </d2e:ncc>

Options

namerequiredconnectiondescription

content-dir
epub-dir

Source Code

        <p:input port="flow">
            <p:pipe port="flow" step="ncc"/>
        </p:input>
        <p:input port="ncc-metadata">
            <p:pipe port="metadata" step="ncc"/>

d2e:container - container

Package the EPUB 3 fileset as a ZIP-file (OCF).

Inputs

nameprimarysequenceconnectiondescription
manifests false result-manifest@manifest
store-complete false

Options

namerequiredconnectiondescription
content-dir
epub-dir
epub-file opf-metadata@metadata

Source Code

<d2e:container name="container" xmlns:p="http://www.w3.org/ns/xproc" xmlns:d2e="http://pipeline.daisy.org/ns/daisy2epub/">
        <p:with-option name="content-dir" select="$content-dir"/>
        <p:with-option name="epub-dir" select="$epub-dir"/>
        <p:with-option name="epub-file" select="concat($output-dir,                                         if (string-length(replace(//dc:title,'[^a-zA-Z0-9_ -]',''))>0)                                             then          replace(//dc:title,'[^a-zA-Z0-9_ -]','')                                             else          'output'                                       ,'.epub')">
            <p:pipe port="opf-metadata" step="metadata"/>
        </p:with-option>
        <p:input port="manifests">
            <p:pipe port="result-manifest" step="manifest"/>
        </p:input>
        <p:input port="store-complete">
            <p:pipe port="store-complete" step="navigation"/>
            <p:pipe port="store-complete" step="media-overlay"/>
            <p:pipe port="store-complete" step="contents"/>
            <p:pipe port="store-complete" step="resources"/>
            <p:pipe port="store-complete" step="package"/>
        </p:input>
    </d2e:container>
